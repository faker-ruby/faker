# Faker::Space

```ruby
# Random planet from our Solar System (8 elements)
Faker::Space.planet #=> "Venus"

# Random moon from our Solar System (20 elements)
Faker::Space.moon #=> "Europa"

# Random galaxy (10 elements)
Faker::Space.galaxy #=> "Andromeda"

# Random nebula name (8 elements)
Faker::Space.nebula #=> "Triffid Nebula"

# Random star cluster (30 elements)
Faker::Space.star_cluster #=> "Messier 70"

# Random constellation (21 elements)
Faker::Space.constellation #=> "Orion"

# Random star (19 elements)
Faker::Space.star #=> "Proxima Centauri"

# Random national space agency (14 elements)
Faker::Space.agency #=> "Japan Aerospace Exploration Agency"

# Random space agency abbreviation (15 elements)
Faker::Space.agency_abv #=> "NASA"

# Random spacecraft name (limited to NASA) (10 elements)
Faker::Space.nasa_space_craft #=> "Endeavour"

# Random private space company title (8 elements)
Faker::Space.company #=> "SpaceX"

# Random unit of stellar distance with number (5 elements)
Faker::Space.distance_measurement #=> "15 parsecs"

# Random meteorite name (162 elements)
Faker::Space.meteorite #=> "Ensisheim"
```
