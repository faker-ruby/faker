def self.fix_umlauts(string)
  string.gsub(/[äöüß]/i) do |match|
    case match.downcase
    when 'ä' then 'ae'
    when 'ö' then 'oe'
    when 'ü' then 'ue'
    when 'ß' then 'ss'
    end
  end
end

def self.prepare(string)
  result = romanize_cyrillic string
  result = fix_umlauts result
  result.gsub(/\W/, '').downcase
end
